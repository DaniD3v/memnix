	hash: OnceLock<hash>,
} : {new -> &'b Self}

ColoredNode {
	& node,
	color
}

// invariant over 'b lifetime (verify)
type SccNode = BTreeMap<*const Node, coloredNode> : {
	new(nodes: Vec<&Node>) {
		nodes.map(|node|
			ColorNode {
				node,
				color: node.hash_initial(nodes),
			}
		)
	}
}
